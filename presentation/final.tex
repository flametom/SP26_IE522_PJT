\begin{frame}[t]{Recap: Simulating Emergency Evacuation}
\small

\begin{columns}[T,onlytextwidth]
\column{0.55\textwidth}

\textbf{Scenario:} A hazard (fire, explosion, flood) strikes a community.\\
Thousands of pedestrians must evacuate simultaneously.

\vspace{0.6em}
\begin{block}{Modeling challenges}
\footnotesize
\begin{itemize}\itemsep0.35em
  \item Agents decide \keyterm{independently}
  \item Decisions use \keyterm{local information}: congestion, hazard proximity
  \item \keyterm{Panic} and congestion create emergent system behavior
\end{itemize}
\end{block}

\vspace{0.45em}
\fcolorbox{highlightblue}{lightbluefill}{\parbox{0.92\linewidth}{\centering\footnotesize
The paper therefore uses \textbf{Agent-Based Simulation}: one autonomous pedestrian per agent.
}}

\column{0.42\textwidth}
\vspace{0.5em}

\begin{columns}[T,onlytextwidth]

\begin{center}
\includegraphics[width=0.96\linewidth]{assets/benchmark_time_dist.pdf}
\end{center}
{\centering\scriptsize Baseline single run: 168s total, 165.9s in path queries.\par}

\column{0.44\textwidth}
\begin{block}{\small Per-agent path computation}
\scriptsize
\begin{itemize}\itemsep0.25em
  \item 2{,}000 agents $\times$ 121 steps
  \item $\sim$242K path queries / run
  \item Same graph traversed repeatedly
\end{itemize}
\end{block}

\vspace{0.45em}
\begin{alertblock}{\small Shared destination trees}
\scriptsize
\begin{itemize}\itemsep0.25em
  \item Group agents by destination
  \item Build one tree per destination
  \item Reconstruct each agent's path from that tree
\end{itemize}
\end{alertblock}
\end{columns}

\vspace{0.7em}

\begin{center}
\fcolorbox{highlightblue}{lightbluefill}{\parbox{0.88\linewidth}{\centering\footnotesize
\textbf{Routing subsystem}: 165.9s $\to$ 20.0s ($\approx$8.3$\times$). \qquad
\textbf{End-to-end}: 168s $\to$ 22.5s ($\approx$7.5$\times$).
}}
\end{center}

\end{frame}

% SLIDE 8: Experimental design and reproducibility
% ============================================================
\begin{frame}[t]{Experimental design for scalable reproduction}
\footnotesize

\begin{block}{\scriptsize 1. Per-step path recomputation}
\scriptsize
\textbf{Implementation decision:} recompute paths every step; no cross-step path caching.\\
\textbf{Rationale:} agents must react to live congestion, otherwise queues stop being emergent.
\end{block}

\vspace{0.3em}

\begin{block}{\scriptsize 2. Parallel experiment execution}
\scriptsize
\textbf{Implementation decision:} run the seed $\times$ config grid with 8 workers.\\
\textbf{Rationale:} 390 runs at the paper's 2.5\,h/run baseline would require \textbf{41 days}; the grid finished in \textbf{3.4 hours}.
\end{block}

\vspace{0.3em}

\begin{block}{\scriptsize 3. Controlled RNG streams}
\scriptsize
\textbf{Implementation decision:} vary agent seeds; keep hazard seed fixed at 1135.\\
\textbf{Rationale:} changes in $P_{\max}$ or $\varepsilon_p$ should not also change the hazard scenario.
\end{block}

\vspace{0.4em}
\begin{center}
\fcolorbox{highlightblue}{lightbluefill}{\parbox{0.92\linewidth}{\centering\scriptsize
These implementation decisions preserve the behavioral logic while making multi-seed reproduction feasible.
}}
\end{center}

\end{frame}
